<script>
    const canvas = document.getElementById('avuruduCanvas');
    const ctx = canvas.getContext('2d');
    
    let uploadedImage = null;
    let selectedFrameIdx = 0;

    // Cultural Festive Theme Presets
    const frames = [
        { name: "Suryan Gold", bg: "#FFFDF0", border: "#D4AF37", accent: "#FF8C00" },
        { name: "Erabadu Red", bg: "#FFF5F3", border: "#B31B04", accent: "#D4AF37" },
        { name: "Royal Teal", bg: "#F4F9F9", border: "#005F73", accent: "#EE9B00" },
        { name: "Traditional Terracotta", bg: "#FAF5F0", border: "#8B4513", accent: "#D4AF37" }
    ];

    // Safe File Upload Input Reader Pipeline
    document.getElementById('imageUpload').addEventListener('change', function(e) {
        const reader = new FileReader();
        reader.onload = function(event) {
            const img = new Image();
            img.onload = function() {
                uploadedImage = img;
                drawCard();
            };
            img.src = event.target.result;
        };
        if(e.target.files[0]) {
            reader.readAsDataURL(e.target.files[0]);
        }
    });

    function changeFrame(index) {
        document.querySelector('.avurudu-item.active').classList.remove('active');
        document.getElementsByClassName('avurudu-item')[index].classList.add('active');
        selectedFrameIdx = index;
        drawCard();
    }

    // Traditional Sun/Star Motif Vector Generator
    function drawSunMotif(ctx, x, y, radius) {
        ctx.save();
        ctx.strokeStyle = frames[selectedFrameIdx].border;
        ctx.fillStyle = frames[selectedFrameIdx].accent;
        ctx.lineWidth = 2;
        
        // Draw Central Sun Disk
        ctx.beginPath();
        ctx.arc(x, y, radius, 0, Math.PI * 2);
        ctx.fill();
        ctx.stroke();

        // Draw Radiating Geometric Rays
        for (let i = 0; i < 12; i++) {
            ctx.save();
            ctx.translate(x, y);
            ctx.rotate((i * Math.PI) / 6);
            ctx.beginPath();
            ctx.moveTo(0, -radius - 2);
            ctx.lineTo(-5, -radius - 12);
            ctx.lineTo(0, -radius - 22);
            ctx.lineTo(5, -radius - 12);
            ctx.closePath();
            ctx.fill();
            ctx.stroke();
            ctx.restore();
        }
        ctx.restore();
    }

    // Core Canvas Processing Engine
    function drawCard() {
        const frame = frames[selectedFrameIdx];
        const greetingText = document.getElementById('userGreeting').value || "";

        // 1. Draw Festive Base Background
        ctx.fillStyle = frame.bg;
        ctx.fillRect(0, 0, 600, 600);

        // 2. Render Sun Backdrop Vector Accents (Traditional Avurudu Symbolism)
        drawSunMotif(ctx, 300, 250, 40);

        // 3. Compute and Composite Smart Square Cropped Photo Over Motif
        if (uploadedImage) {
            ctx.save();
            
            const imgX = 130;
            const imgY = 85;
            const imgSize = 340;

            // Circular frame cut option for cultural aesthetics
            ctx.beginPath();
            ctx.arc(300, 255, 170, 0, Math.PI * 2);
            ctx.clip();

            // Mathematical Aspect-Ratio Safe Scaling Vector
            const minSide = Math.min(uploadedImage.width, uploadedImage.height);
            const sx = (uploadedImage.width - minSide) / 2;
            const sy = (uploadedImage.height - minSide) / 2;

            ctx.drawImage(uploadedImage, sx, sy, minSide, minSide, imgX, imgY, imgSize, imgSize);
            ctx.restore();

            // Circular Outer Ring Frame Line Overlay
            ctx.strokeStyle = frame.border;
            ctx.lineWidth = 5;
            ctx.beginPath();
            ctx.arc(300, 255, 171, 0, Math.PI * 2);
            ctx.stroke();
        }

        // 4. Draw Traditional Intricate Decorative Borders
        ctx.strokeStyle = frame.border;
        ctx.lineWidth = 6;
        ctx.strokeRect(25, 25, 550, 550);
        
        ctx.strokeStyle = frame.accent;
        ctx.lineWidth = 2;
        ctx.strokeRect(34, 34, 532, 532);

        // Intricate interlocking corner points
        ctx.fillStyle = frame.border;
        const pts = [25, 555];
        pts.forEach(x => {
            pts.forEach(y => {
                ctx.beginPath();
                ctx.arc(x == 25 ? x+10 : x-10, y == 25 ? y+10 : y-10, 14, 0, Math.PI * 2);
                ctx.fill();
            });
        });

        // 5. Native Typography & Wrapping Compiler
        ctx.textAlign = 'center';
        ctx.fillStyle = frame.border;
        ctx.font = 'bold 22px "Segoe UI", Arial, sans-serif';

        const words = greetingText.split(' ');
        let line = '';
        let startY = 495;
        const lineHeight = 32;

        for (let n = 0; n < words.length; n++) {
            let testLine = line + words[n] + ' ';
            let metrics = ctx.measureText(testLine);
            if (metrics.width > 480 && n > 0) {
                ctx.fillText(line, 300, startY);
                line = words[n] + ' ';
                startY += lineHeight;
            } else {
                line = testLine;
            }
        }
        ctx.fillText(line, 300, startY);
    }

    // Live update dynamic state triggers
    document.getElementById('userGreeting').addEventListener('input', drawCard);

    // File Extraction Save Download Triggers
    document.getElementById('downloadAvuruduBtn').addEventListener('click', function() {
        const link = document.createElement('a');
        link.download = `Sinhala_Tamil_NewYear_Card.png`;
        link.href = canvas.toDataURL('image/png');
        link.click();
    });

    // Handle instant app initialization mounting loops safely
    window.onload = () => { drawCard(); };
    setTimeout(drawCard, 300);
</script>
